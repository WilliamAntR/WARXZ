import torch
import numpy as np

def cutmix_data(x, y, alpha=1.0):
    """
    x : batch gambar  [B, C, H, W]
    y : batch label   [B]
    alpha : parameter distribusi Beta
    """
    if alpha > 0:
        lam = np.random.beta(alpha, alpha)
    else:
        lam = 1.0

    batch_size = x.size(0)
    H = x.size(2)
    W = x.size(3)

    # Shuffle index
    index = torch.randperm(batch_size).to(x.device)

    # Hitung ukuran kotak yang akan dipotong
    cut_ratio = np.sqrt(1.0 - lam)
    cut_h = int(H * cut_ratio)
    cut_w = int(W * cut_ratio)

    # Pilih titik tengah secara acak
    cx = np.random.randint(W)
    cy = np.random.randint(H)

    # Hitung koordinat kotak (pastikan tidak keluar batas)
    x1 = max(0, cx - cut_w // 2)
    x2 = min(W, cx + cut_w // 2)
    y1 = max(0, cy - cut_h // 2)
    y2 = min(H, cy + cut_h // 2)

    # Tempelkan potongan gambar B ke gambar A
    mixed_x = x.clone()
    mixed_x[:, :, y1:y2, x1:x2] = x[index, :, y1:y2, x1:x2]

    # Hitung ulang lambda berdasarkan area aktual
    lam = 1 - (y2 - y1) * (x2 - x1) / (H * W)

    y_a, y_b = y, y[index]

    return mixed_x, y_a, y_b, lam


def cutmix_criterion(criterion, pred, y_a, y_b, lam):
    """Loss function untuk CutMix"""
    return lam * criterion(pred, y_a) + (1 - lam) * criterion(pred, y_b)
